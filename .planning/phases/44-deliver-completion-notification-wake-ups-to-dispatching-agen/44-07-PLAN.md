    ```ts
    /**
     * Phase 44 D-44-RECOVERY: boot-time replay of unnotified terminal subscriptions.
     *
     * Fires wake-ups for active subscriptions whose task is already in a terminal status
     * AND whose notifiedStatuses ledger does NOT yet record that terminal status. Guards
     * against daemon crashes between transition() and plugin ACK.
     *
     * Safe to call multiple times — each call observes the same notifiedStatuses ledger,
     * so the second call short-circuits via the existing .includes(toStatus) filter.
     */
    async replayUnnotifiedTerminals(store: ITaskStore): Promise<void> {
      const subscriptionStore = createSubscriptionStore(store);
      const allTasks = await store.listAll(); // confirm actual method name via reading task-store.ts
      let replayed = 0;

      for (const task of allTasks) {
        const toStatus = task.frontmatter.status;
        if (!TERMINAL_STATUSES.has(toStatus)) continue;

        const active = await subscriptionStore.list(task.frontmatter.id, { status: "active" });
        const candidates = active.filter(
          (s) =>
            resolveDeliveryKind(s) === OPENCLAW_CHAT_DELIVERY_KIND &&
            !s.notifiedStatuses.includes(toStatus),
        );

        for (const sub of candidates) {
          await this.deliverOne(store, subscriptionStore, task, sub, toStatus, "recovery");
          replayed += 1;
        }
      }

      wakeLog.info({ replayed }, "wake-up.recovery-pass-complete");
    }
    ```
