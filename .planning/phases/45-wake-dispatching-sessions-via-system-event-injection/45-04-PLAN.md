    ```typescript
    /**
     * Phase 45 (D-45-MESSAGE-BREVITY + D-45-BUG-AGENT-UNKNOWN + D-45-FALLBACK-WARNING)
     *
     * One-line chat audit message:
     *   ✓ TASK-NNN (done|review) — title [agent: <id>]      ← success-class glyph
     *   ⚠ TASK-NNN (failed|cancelled|deadletter) — title [agent: <id>]   ← failure-class
     *
     * When systemEventCapable=false, the inline FALLBACK_WARNING_LINE is appended
     * on a second line so the human user sees the capability gap in chat.
     *
     * Agent id is sourced from delivery.dispatcherAgentId (Phase 44 captured this
     * during dispatch). Falls back to "?" when undefined. NEVER renders
     * "Agent: unknown" — that was the Phase 44 bug fixed here.
     */
    function renderMessage(args: {
      task: NonNullable<Awaited<ReturnType<ITaskStore["get"]>>>;
      toStatus: string;
      dispatcherAgentId: string | undefined;
      systemEventUnavailable: boolean;
    }): string {
      const { task, toStatus, dispatcherAgentId, systemEventUnavailable } = args;
      const glyph = renderStatusGlyph(toStatus);
      const id = task.frontmatter.id;
      const title = truncate(String(task.frontmatter.title ?? ""), 120);
      const agent = dispatcherAgentId ?? "?";
      let line = `${glyph} ${id} (${toStatus}) — ${title} [agent: ${agent}]`;
      if (systemEventUnavailable) {
        line += `\n${FALLBACK_WARNING_LINE}`;
      }
      return line;
    }

    function renderStatusGlyph(status: string): string {
      // ✓ for success-class transitions (done, review); ⚠ for failure-class
      // (failed, cancelled, deadletter, anything else).
      return status === "done" || status === "review" ? "✓" : "⚠";
    }
    ```
