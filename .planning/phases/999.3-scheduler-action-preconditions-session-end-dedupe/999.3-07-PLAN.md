       import { createTestHarness } from "../../src/testing/harness.js";
       import { handleStaleHeartbeat } from "../../src/dispatch/recovery-handlers.js";
       import { InMemoryTaskLockManager } from "../../src/protocol/task-lock.js";

       describe("Phase 999.3 — integration sweep (L13)", () => {
         it("end-to-end: stale_heartbeat after status moved emits scheduler_action_skipped, NOT scheduler_action_failed", async () => {
           const h = await createTestHarness();
           try {
             const lockManager = new InMemoryTaskLockManager();
             const config = { dataDir: h.dataDir, dryRun: false, defaultLeaseTtlMs: 60_000, lockManager };

             const created = await h.store.create({ /* fixture */ } as any);
             await h.store.transition(created.frontmatter.id, "in-progress");
             const queueTimestamp = (await h.store.get(created.frontmatter.id))!.frontmatter.lastTransitionAt;
             await h.store.transition(created.frontmatter.id, "blocked");

             await handleStaleHeartbeat(
               {
                 type: "stale_heartbeat",
                 taskId: created.frontmatter.id,
                 taskTitle: created.frontmatter.title,
                 reason: "test",
                 expected: { status: "in-progress", leaseAgent: "agent-1", lastTransitionAtBefore: queueTimestamp },
               },
               h.store, h.logger, config,
             );

             const events = await h.readEvents();
             const skipped = events.filter((e) => e.type === "scheduler_action_skipped");
             const failed = events.filter((e) => e.type === "scheduler_action_failed" && (e.payload as any).type === "stale_heartbeat");
             expect(skipped.length).toBeGreaterThanOrEqual(1);
             expect(failed.length).toBe(0);
           } finally {
             await h.cleanup();
           }
         });
       });
