       ```ts
       export async function handleSlaViolation(
         action: SchedulerAction,
         store: ITaskStore,
         logger: EventLogger,
         config: SchedulerConfig,
       ): Promise<ActionHandlerResult> {
         const body = async (): Promise<ActionHandlerResult> => {
           const task = await store.get(action.taskId);
           if (!task) {
             log.warn({ taskId: action.taskId, op: "slaViolation" }, "sla_violation: task not found, skipping");
             await logSkippedSafely(logger, action, "handleSlaViolation", null, "task_gone");
             return { executed: false, failed: false };
           }
           const match = matchesExpected(task, action.expected);
           if (!match.ok) {
             log.info(
               { taskId: action.taskId, handler: "handleSlaViolation", expected: action.expected, actual: match.actual, reason: match.reason, op: "preconditionStale" },
               "scheduler action skipped — precondition no longer holds",
             );
             await logSkippedSafely(logger, action, "handleSlaViolation", match.actual, match.reason);
             return { executed: false, failed: false };
           }
           // … existing handleSlaViolation body unchanged from this point on …
         };

         return config.lockManager
           ? config.lockManager.withLock(action.taskId, body)
           : body();
       }
       ```
