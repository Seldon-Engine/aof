NEW src/dispatch/precondition.ts (per RESEARCH.md §"Implementation Sketch"):
```ts
import type { Task, TaskStatus } from "../schemas/task.js";
import type { SchedulerAction } from "./types.js";
import type { EventLogger } from "../events/logger.js";
import { createLogger } from "../logging/index.js";

const log = createLogger("precondition");

export interface PreconditionActual {
  status: TaskStatus;
  leaseAgent: string | null;
  lastTransitionAt: string;
}

export type MatchReason =
  | "status_mismatch"
  | "leaseAgent_mismatch"
  | "lastTransitionAt_advanced";

export type MatchResult =
  | { ok: true }
  | { ok: false; reason: MatchReason; actual: PreconditionActual };

export function matchesExpected(
  task: Task,
  expected: SchedulerAction["expected"] | undefined,
): MatchResult {
  if (!expected) return { ok: true };
  const actual: PreconditionActual = {
    status: task.frontmatter.status,
    leaseAgent: task.frontmatter.lease?.agent ?? null,
    lastTransitionAt: task.frontmatter.lastTransitionAt,
  };
  // Order: status → leaseAgent → lastTransitionAtBefore. Short-circuit on first mismatch.
  if (expected.status !== undefined && expected.status !== actual.status) {
    return { ok: false, reason: "status_mismatch", actual };
  }
  if (expected.leaseAgent !== undefined && expected.leaseAgent !== actual.leaseAgent) {
    return { ok: false, reason: "leaseAgent_mismatch", actual };
  }
  if (
    expected.lastTransitionAtBefore !== undefined &&
    actual.lastTransitionAt > expected.lastTransitionAtBefore
  ) {
    return { ok: false, reason: "lastTransitionAt_advanced", actual };
  }
  return { ok: true };
}

export async function logSkippedSafely(
  logger: EventLogger,
  action: SchedulerAction,
  handler: string,
  actual: PreconditionActual | null,
  reason: "task_gone" | MatchReason,
): Promise<void> {
  try {
    await logger.log("scheduler_action_skipped", "scheduler", {
      taskId: action.taskId,
      payload: {
        handler,
        type: action.type,
        expected: action.expected ?? null,
        actual,
        reason,
      },
    });
  } catch (err) {
    log.warn(
      { err, taskId: action.taskId, op: "eventLog" },
      "event logger write failed (best-effort)",
    );
  }
}
```
