    ```typescript
    import { EventEmitter } from "node:events";
    import { randomUUID } from "node:crypto";
    import { createLogger } from "../logging/index.js";
    import type { SpawnRequest } from "./schemas.js";

    const log = createLogger("spawn-queue");

    export class SpawnQueue extends EventEmitter {
      private pending = new Map<string, SpawnRequest>();
      private claimed = new Set<string>();

      enqueue(partial: Omit<SpawnRequest, "id">): SpawnRequest {
        const full: SpawnRequest = { id: randomUUID(), ...partial } as SpawnRequest;
        this.pending.set(full.id, full);
        this.emit("enqueue", full);
        log.debug({ id: full.id, taskId: full.taskId }, "spawn enqueued");
        return full;
      }

      claim(): SpawnRequest | undefined {
        for (const [id, sr] of this.pending) {
          if (!this.claimed.has(id)) {
            this.claimed.add(id); this.pending.delete(id);
            return sr;
          }
        }
        return undefined;
      }

      tryClaim(id: string): boolean {
        if (!this.pending.has(id)) return false;
        this.claimed.add(id); this.pending.delete(id);
        return true;
      }

      releaseClaim(id: string, sr: SpawnRequest): void {
        this.claimed.delete(id);
        this.pending.set(id, sr);
      }

      reset(): void { this.pending.clear(); this.claimed.clear(); this.removeAllListeners(); }
    }
    ```

    ```typescript
    import { createLogger } from "../logging/index.js";

    const log = createLogger("plugin-registry");

    export interface PluginHandle { pluginId: string; connectedAt: number; release(): void; }

    export class PluginRegistry {
      private active = new Map<string, PluginHandle>();

      register(req: IncomingMessage | NodeJS.EventEmitter, res: ServerResponse | NodeJS.EventEmitter, pluginId: string = "openclaw"): PluginHandle {
        const handleId = `${pluginId}-${Date.now()}-${Math.random().toString(36).slice(2)}`;
        const handle: PluginHandle = {
          pluginId, connectedAt: Date.now(),
          release: () => {
            if (this.active.has(handleId)) { this.active.delete(handleId); log.info({ pluginId }, "plugin detached"); }
          },
        };
        this.active.set(handleId, handle);
        log.info({ pluginId }, "plugin attached");
        res.on("close", () => handle.release());
        return handle;
      }

      hasActivePlugin(pluginId: string = "openclaw"): boolean {
        for (const h of this.active.values()) if (h.pluginId === pluginId) return true;
        return false;
      }

      activeCount(): number { return this.active.size; }
      reset(): void { this.active.clear(); }
    }
    ```

    ```typescript
    export const handleSpawnWait: RouteHandler = async (req, res, deps) => {
      if (!deps.spawnQueue || !deps.pluginRegistry) { /* 500 unavailable */ return; }
      const handle = deps.pluginRegistry.register(req, res);

      const claimant = deps.spawnQueue.claim();
      if (claimant) { res.writeHead(200, { "Content-Type": "application/json" }); res.end(JSON.stringify(claimant)); return; }

      const KEEPALIVE_MS = 25_000;
      let settled = false;
      const onEnqueue = (sr: SpawnRequest) => {
        if (settled) return;
        if (!deps.spawnQueue!.tryClaim(sr.id)) return;  // another waiter claimed first
        settled = true;
        clearTimeout(timer);
        deps.spawnQueue!.off("enqueue", onEnqueue);
        res.writeHead(200, { "Content-Type": "application/json" });
        res.end(JSON.stringify(sr));
      };
      const timer = setTimeout(() => {
        if (settled) return;
        settled = true;
        deps.spawnQueue!.off("enqueue", onEnqueue);
        res.writeHead(204); res.end();
      }, KEEPALIVE_MS);
      res.on("close", () => {
        if (settled) return;
        settled = true;
        clearTimeout(timer);
        deps.spawnQueue!.off("enqueue", onEnqueue);
      });
      deps.spawnQueue.on("enqueue", onEnqueue);
    };
    ```

    ```typescript
    import { createLogger } from "../logging/index.js";
    import type { GatewayAdapter, TaskContext, SpawnResult, SessionStatus, AgentRunOutcome } from "./executor.js";
    import type { SpawnQueue } from "../ipc/spawn-queue.js";
    import type { PluginRegistry } from "../ipc/plugin-registry.js";
    import type { SpawnResultPost } from "../ipc/schemas.js";

    const log = createLogger("plugin-bridge-adapter");

    export class PluginBridgeAdapter implements GatewayAdapter {
      private pending = new Map<string, { taskId: string; onRunComplete?: (o: AgentRunOutcome) => void | Promise<void> }>();

      constructor(private readonly queue: SpawnQueue, private readonly registry: PluginRegistry) {}

      async spawnSession(context: TaskContext, opts?: { timeoutMs?: number; correlationId?: string; onRunComplete?: (o: AgentRunOutcome) => void | Promise<void> }): Promise<SpawnResult> {
        if (!this.registry.hasActivePlugin()) return { success: false, error: "no-plugin-attached" };

        const sr = this.queue.enqueue({
          taskId: context.taskId,
          taskPath: context.taskPath,
          agent: context.agent,
          priority: context.priority,
          thinking: context.thinking as string | undefined,
          routing: context.routing,
          projectId: context.projectId,
          projectRoot: context.projectRoot,
          taskRelpath: context.taskRelpath,
          timeoutMs: opts?.timeoutMs ?? context.timeoutMs,
          correlationId: opts?.correlationId,
          callbackDepth: (context.metadata?.callbackDepth as number) ?? 0,
        });
        this.pending.set(sr.id, { taskId: context.taskId, onRunComplete: opts?.onRunComplete });
        log.info({ spawnId: sr.id, taskId: context.taskId, agent: context.agent }, "spawn enqueued for plugin");
        return { success: true, sessionId: sr.id };
      }

      async deliverResult(spawnId: string, result: SpawnResultPost): Promise<void> {
        const rec = this.pending.get(spawnId);
        this.pending.delete(spawnId);
        if (!rec) { log.warn({ spawnId }, "deliverResult: no pending entry — ignoring"); return; }
        if (!rec.onRunComplete) return;
        try {
          await rec.onRunComplete({
            taskId: rec.taskId,
            sessionId: result.sessionId,
            success: result.success,
            aborted: result.aborted,
            error: result.error,
            durationMs: result.durationMs,
          });
        } catch (err) {
          log.error({ err, spawnId, taskId: rec.taskId }, "onRunComplete callback threw");
        }
      }

      async getSessionStatus(sessionId: string): Promise<SessionStatus> {
        return { sessionId, alive: this.pending.has(sessionId) };
      }

      async forceCompleteSession(sessionId: string): Promise<void> {
        const rec = this.pending.get(sessionId);
        this.pending.delete(sessionId);
        if (rec?.onRunComplete) {
          await rec.onRunComplete({ taskId: rec.taskId, sessionId, success: false, aborted: true, durationMs: 0 });
        }
      }
    }
    ```

    ```typescript
    import type { GatewayAdapter, TaskContext, SpawnResult, SessionStatus, AgentRunOutcome } from "./executor.js";
    import type { PluginRegistry } from "../ipc/plugin-registry.js";

    const log = createLogger("selecting-adapter");

    export interface SelectingAdapterOpts {
      primary: GatewayAdapter;
      fallback: GatewayAdapter;
      registry: PluginRegistry;
      mode: "plugin-bridge" | "standalone";
    }

    export class SelectingAdapter implements GatewayAdapter {
      constructor(private readonly opts: SelectingAdapterOpts) {}

      async spawnSession(context: TaskContext, spawnOpts?: Parameters<GatewayAdapter["spawnSession"]>[1]): Promise<SpawnResult> {
        if (this.opts.registry.hasActivePlugin()) return this.opts.primary.spawnSession(context, spawnOpts);
        if (this.opts.mode === "standalone") return this.opts.fallback.spawnSession(context, spawnOpts);
        log.info({ taskId: context.taskId, op: "hold" }, "holding task: no-plugin-attached");
        return { success: false, error: "no-plugin-attached" };
      }
    ```
