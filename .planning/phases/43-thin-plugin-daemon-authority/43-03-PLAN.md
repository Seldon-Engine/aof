    ```typescript
    import type { ITaskStore } from "../store/interfaces.js";
    import type { EventLogger } from "../events/logger.js";
    import { createProjectStore } from "../projects/store-factory.js";
    import { loadOrgChart } from "../org/loader.js";
    import { PermissionAwareTaskStore } from "../permissions/task-permissions.js";
    import type { OrgChart } from "../schemas/org-chart.js";
    import { createLogger } from "../logging/index.js";

    const log = createLogger("ipc-store-resolver");

    export interface StoreResolverOpts {
      dataDir: string;
      baseStore: ITaskStore;
      logger: EventLogger;
      orgChartPath?: string;
    }

    export function buildDaemonResolveStore(opts: StoreResolverOpts): (req: { actor?: string; projectId?: string }) => Promise<ITaskStore> {
      const projectStores = new Map<string, ITaskStore>();
      const initialized = new WeakSet<ITaskStore>();
      let orgChartPromise: Promise<OrgChart | undefined> | undefined;
      if (opts.orgChartPath) {
        orgChartPromise = loadOrgChart(opts.orgChartPath).then(r => r.success && r.chart ? r.chart : undefined).catch(() => undefined);
      }

      async function ensureInit(s: ITaskStore): Promise<ITaskStore> {
        if (initialized.has(s)) return s;
        if ("init" in s && typeof s.init === "function") await s.init();
        initialized.add(s);
        return s;
      }

      async function resolveProjectStore(projectId?: string): Promise<ITaskStore> {
        if (!projectId) return ensureInit(opts.baseStore);
        const cached = projectStores.get(projectId);
        if (cached) return ensureInit(cached);
        const { store } = await createProjectStore({ projectId, vaultRoot: opts.dataDir, logger: opts.logger });
        projectStores.set(projectId, store);
        return ensureInit(store);
      }

      return async ({ actor, projectId }) => {
        const base = await resolveProjectStore(projectId);
        if (!orgChartPromise || !actor || actor === "unknown") return base;
        const chart = await orgChartPromise;
        if (!chart) return base;
        return new PermissionAwareTaskStore(base, chart, actor);
      };
    }
    ```

    ```typescript
      server.keepAliveTimeout = 60_000;
      server.headersTimeout = 61_000;

      server.on("request", (req, res) => {
        if (!req.url?.startsWith("/v1/")) return;  // let existing /healthz, /status handlers through
        void (async () => {
          try {
            if (req.method === "POST" && req.url === "/v1/tool/invoke") return await handleInvokeTool(req, res, deps);
            if (req.method === "POST" && req.url === "/v1/event/session-end") return await handleSessionEnd(req, res, deps);
            if (req.method === "POST" && req.url === "/v1/event/agent-end") return await handleAgentEnd(req, res, deps);
            if (req.method === "POST" && req.url === "/v1/event/before-compaction") return await handleBeforeCompaction(req, res, deps);
            if (req.method === "POST" && req.url === "/v1/event/message-received") return await handleMessageReceived(req, res, deps);
            // Wave 2 adds /v1/spawns/wait and /v1/spawns/{id}/result here.
            res.writeHead(404, { "Content-Type": "application/json" });
            res.end(JSON.stringify({ error: { kind: "not-found", message: `no route: ${req.method} ${req.url}` } }));
          } catch (err) {
            deps.log.error({ err, url: req.url }, "IPC route threw");
            if (!res.headersSent) { res.writeHead(500, { "Content-Type": "application/json" }); res.end(JSON.stringify({ error: { kind: "internal", message: (err as Error).message } })); }
          }
        })();
      });
    }
    ```
