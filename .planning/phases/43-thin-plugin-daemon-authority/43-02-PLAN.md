    Create `tests/integration/helpers/plugin-ipc-client.ts`:
    ```typescript
    import { request as httpRequest } from "node:http";

    // Local minimal types so helper compiles without the Wave 1 src/ipc/schemas.ts.
    export interface InvokeToolEnvelope {
      pluginId?: string; name: string; params: Record<string, unknown>;
      actor?: string; projectId?: string; correlationId?: string;
      toolCallId: string; callbackDepth?: number;
    }

    async function postJson<T>(socketPath: string, path: string, body: unknown, timeoutMs = 10000): Promise<T> {
      return new Promise((resolve, reject) => {
        const payload = JSON.stringify(body);
        const req = httpRequest(
          { socketPath, path, method: "POST", timeout: timeoutMs,
            headers: { "Content-Type": "application/json", "Content-Length": Buffer.byteLength(payload) } },
          (res) => { let data = ""; res.on("data", c => data += c); res.on("end", () => { try { resolve(JSON.parse(data)); } catch (e) { reject(e); } }); }
        );
        req.on("error", reject);
        req.on("timeout", () => { req.destroy(new Error("timeout")); });
        req.write(payload); req.end();
      });
    }

    export async function invokeTool(socketPath: string, envelope: InvokeToolEnvelope) {
      return postJson<{ result?: unknown; error?: { kind: string; message: string } }>(socketPath, "/v1/tool/invoke", envelope);
    }

    export async function waitForSpawn(socketPath: string, timeoutMs = 30000): Promise<unknown | undefined> {
      // GET /v1/spawns/wait — resolves undefined on 204
      return new Promise((resolve, reject) => {
        const req = httpRequest({ socketPath, path: "/v1/spawns/wait", method: "GET", timeout: timeoutMs + 5000 }, (res) => {
          if (res.statusCode === 204) { res.resume(); return resolve(undefined); }
          let data = ""; res.on("data", c => data += c); res.on("end", () => { try { resolve(JSON.parse(data)); } catch (e) { reject(e); } });
        });
        req.on("error", reject); req.on("timeout", () => req.destroy(new Error("timeout")));
        req.end();
      });
    }

    export async function postSpawnResult(socketPath: string, id: string, result: unknown) {
      return postJson(socketPath, `/v1/spawns/${encodeURIComponent(id)}/result`, result);
    }

    export async function postEvent(socketPath: string, path: string, body: unknown) {
      return postJson(socketPath, path, body);
    }
    ```
