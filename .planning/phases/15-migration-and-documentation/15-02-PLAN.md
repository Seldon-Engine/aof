export const Hop = z.object({
  id: z.string(),
  role: z.string(),
  dependsOn: z.array(z.string()).default([]),
  condition: ConditionExpr.optional(),
  autoAdvance: z.boolean().default(true),

  rejectionStrategy: RejectionStrategy.optional(),

  timeout: z.string().optional(),
  escalateTo: z.string().optional(),

export const WorkflowDefinition = z.object({
  name: z.string(),
  hops: z.array(Hop).min(1),
});

export const HopStatus = z.enum(["pending", "ready", "dispatched", "complete", "failed", "skipped"]);

export const TaskWorkflow = z.object({
  definition: WorkflowDefinition,
  state: WorkflowState,
  templateName: z.string().optional(),
