```typescript
async function loadProjectManifest(
  store: ITaskStore,
  projectId: string
): Promise<ProjectManifest | null> {
  try {
    const projectPath = join(store.projectRoot, "projects", projectId, "project.yaml");
    const content = await readFile(projectPath, "utf-8");
    const manifest = parseYaml(content) as ProjectManifest;
    return manifest;
  } catch {
    return null;
```

**2. Wire project-scoped store resolution in adapter (src/openclaw/adapter.ts):**

a) Modify `registerAofPlugin()` to accept a way to resolve project stores. Add a `projectStores` parameter to `AOFPluginOptions`:
```typescript
export interface AOFPluginOptions {
  // ... existing fields ...
  /** Map of project ID -> task store for multi-project resolution */
  projectStores?: Map<string, ITaskStore>;
}
```

b) Inside `registerAofPlugin()`, create a `resolveProjectStore()` helper function:
```typescript
const resolveProjectStore = (projectId?: string): ITaskStore => {
  if (projectId && opts.projectStores?.has(projectId)) {
    return opts.projectStores.get(projectId)!;
  }
  return store; // Fall back to global store for backward compat
};
```

c) Update all tool execute handlers to extract the `project` param (if provided) and resolve the project-scoped store. The pattern for each tool becomes:
```typescript
execute: async (_id: string, params: Record<string, unknown>) => {
  const actor = (params as any).actor;
  const projectId = (params as any).project as string | undefined;
  const projectStore = resolveProjectStore(projectId);
  const permissionStore = await getStoreForActor(actor, projectStore);
  const result = await aofDispatch({ store: permissionStore, logger, projectId }, params as any);
  return wrapResult(result);
},
```

d) Update `getStoreForActor()` to accept a base store parameter (defaulting to the global store):
```typescript
const getStoreForActor = async (actor?: string, baseStore?: ITaskStore): Promise<ITaskStore> => {
  const effectiveStore = baseStore ?? store;
  if (!orgChartPromise || !actor || actor === "unknown") {
    return effectiveStore;
  }
  const orgChart = await orgChartPromise;
  if (!orgChart) {
    return effectiveStore;
  }
  return new PermissionAwareTaskStore(effectiveStore, orgChart, actor);
};
```

**1. Add participant filtering in buildDispatchActions() (src/dispatch/task-dispatcher.ts):**

After the existing routing target resolution (`const targetAgent = routing.agent ?? ...`) and before pushing the assign action, add participant checking:

```typescript
// PROJ-03: Check project participant list before assigning
const projectId = task.frontmatter.project;
if (projectId && targetAgent) {
  const manifest = await loadProjectManifest(store, projectId);
  if (manifest?.participants && manifest.participants.length > 0) {
    if (!manifest.participants.includes(targetAgent)) {
      actions.push({
        type: "alert",
```
